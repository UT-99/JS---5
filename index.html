        var Count_Occ = (s) => 
            {
              let freeArr = new Array(100);
              freeArr.fill(0);
              let res= '';
              for (let i =0 ; i < s.length ; i++){
                freeArr[s[i].charCodeAt()-97]++;
                
              }
              
              for (let j = 0 ; j < 26 ; j++){
                if(freeArr[j] > 1){
                  res += String.fromCharCode(j+97);
                  res += freeArr[j];
                }
              }
              return res;
            };

Ques. 9. Reverse the String 

        var Reverse_String = (str) => 
            {
                 newstr ='';
                 for (let i = str.length-1 ; i >=0 ; i--){
                   newstr = newstr + str.charAt(i);
                 }
                 return newstr;
            };
